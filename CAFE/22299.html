const bot = BotManager.getCurrentBot();

const Yacht = {

SCORE: ["aces", "deuces", "threes", "fours", "fives", "sixes",

"choice", "foak", "fh", "ss", "ls", "yacht"],

made: false,

maker: "",

started: false,

players: [],

turn: 0,

realTurn: 0,

Player(name) {

this.name = name;

this.scoreBoard = new Yacht.PlayerScoreBoard();

this.diceKeeper = [];

this.diceResult = [];

this.diceOrder = 0;

},

PlayerScoreBoard() {

this.aces = null;

this.deuces = null;

this.threes = null;

this.fours = null;

this.fives = null;

this.sixes = null;

this.bonus = 0;

this.choice = null;

this.foak = null;

this.fh = null;

this.ss = null;

this.ls = null;

this.yacht = null;

this.total = 0;

},

init() {

Yacht.made = false;

Yacht.maker = "";

Yacht.started = false;

Yacht.players = [];

Yacht.turn = 0;

Yacht.realTurn = 0;

},

rollDices(count) {

let result = new Array(count).fill(0).map(() => Math.floor(Math.random() * 6) + 1);

return result;

},

rollDice(player) {

let dice = [];

switch(player.diceOrder) {

case 0:

dice = Yacht.rollDices(5);

player.diceResult = dice;

player.diceOrder++;

break;

case 1:

case 2:

dice = Yacht.rollDices(player.diceResult.length);

player.diceResult = dice;

player.diceOrder++;

break;

}

return dice;

},

keepDice(player, dices) {

for (let i = 0; i < dices.length; i++) {

player.diceResult.splice(player.diceResult.indexOf(dices[i]), 1);

}

player.diceKeeper = player.diceKeeper.concat(dices);

},

discardDice(player, dices) {

player.diceResult = player.diceResult.concat(dices);

for (let i = 0; i < dices.length; i++) {

player.diceKeeper.splice(player.diceKeeper.indexOf(dices[i]), 1);

}

},

recodeDices(player, scoreIndex) {

Yacht.discardDice(player, player.diceKeeper.slice());

player.diceResult.sort((a, b) => a - b);

let score = 0;

scoreIndex--;

if (player.scoreBoard[Yacht.SCORE[scoreIndex]] != null) {

return false;

}

Log.d(JSON.stringify(player));

switch (scoreIndex) {

case 0:

case 1:

case 2:

case 3:

case 4:

case 5:

score = player.diceResult.filter(_ => _ == scoreIndex + 1).length * 6;

break;

case 6:

player.diceResult.forEach(_ => score += _);

break;

case 7:

let fourBool = false;

four_loop: for (let i = 1; i < 7; i++) {

if (player.diceResult.filter(_ => _ == i).length >= 4) {

fourBool = true;

break four_loop;

}

}

if (fourBool) {

player.diceResult.forEach(_ => score += _);

}

break;

case 8:

let fhBool = false;

if (player.diceResult[1] == player.diceResult[2]) {

let fhCondition_1 = player.diceResult[0] == player.diceResult[1];

let fhCondition_2 = player.diceResult[3] == player.diceResult[4];

if (fhCondition_1 && fhCondition_2) fhBool = true;

} else {

let fhCondition_1 = player.diceResult[0] == player.diceResult[1];

let fhCondition_2 = player.diceResult[2] == player.diceResult[3] && player.diceResult[3] == player.diceResult[4];

if (fhCondition_1 && fhCondition_2) fhBool = true;

}

if (fhBool) {

player.diceResult.forEach(_ => score += _);

}

break;

case 9:

let s_dices = Array.from(new Set(player.diceResult));

let ssCondition_1 = (s_dices[0] + 1 == s_dices[1]) && (s_dices[1] + 1 == s_dices[2]) && (s_dices[2] + 1 == s_dices[3]);

let ssCondition_2 = (s_dices[1] + 1 == s_dices[2]) && (s_dices[2] + 1 == s_dices[3]) && (s_dices[3] + 1 == s_dices[4]);

if (ssCondition_1 || ssCondition_2) score = 15;

break;

case 10:

let l_dices = player.diceResult;

if ((l_dices[0] + 1 == l_dices[1]) && (l_dices[1] + 1 == l_dices[2]) &&

(l_dices[2] + 1 == l_dices[3]) && (l_dices[3] + 1 == l_dices[4])) {

score = 30;

}

break;

case 11:

if (Array.from(new Set(player.diceResult)).length == 1) {

score = 50;

}

break;

}

player.scoreBoard[Yacht.SCORE[scoreIndex]] = score;

if (player.scoreBoard.aces + player.scoreBoard.deuces + player.scoreBoard.threes + player.scoreBoard.fours + player.scoreBoard.fives + player.scoreBoard.sixes >= 63) {

if (player.scoreBoard.bonus != 0) player.scoreBoard.bonus = 35;

}

player.scoreBoard.total = 0;

for (let i = 0; i < Yacht.SCORE.length; i++) {

player.scoreBoard.total += player.scoreBoard[Yacht.SCORE[i]];

}

player.diceOrder = 0;

player.diceKeeper = [];

player.diceResult = [];

return true;

},

printNowDice(player) {

return "현재 턴: " + (Yacht.realTurn + 1)

+ "\n주사위: [ " + player.diceResult.join(" ] [ ") + " ]"

+ "\n보관한 주사위: [ " + player.diceKeeper.join(" ] [ ") + " ]"

+ "\n남은 굴리기 기회: " + (3 - player.diceOrder)

+ "\n'!!y help' 로 명령어 도움말을 보실 수 있습니다."

+ Yacht.printNowScore(player);

},

printNowScore(player) {

return ("\u200b".repeat(500) + "\n"

+ "\n[ 족보번호 ] 족보명: 족보점수\n"

+ "\n[ 1 ] Aces: " + player.scoreBoard.aces

+ "\n[ 2 ] Deuces: " + player.scoreBoard.deuces

+ "\n[ 3 ] Threes: " + player.scoreBoard.threes

+ "\n[ 4 ] Fours: " + player.scoreBoard.fours

+ "\n[ 5 ] Fives: " + player.scoreBoard.fives

+ "\n[ 6 ] Sixes: " + player.scoreBoard.sixes

+ "\nBonus: " + player.scoreBoard.bonus + "\n"

+ "\n[ 7 ] Choice: " + player.scoreBoard.choice

+ "\n[ 8 ] 4 of a Kind: " + player.scoreBoard.foak

+ "\n[ 9 ] Full House: " + player.scoreBoard.fh

+ "\n[ 10 ] Small Straight: " + player.scoreBoard.ss

+ "\n[ 11 ] Large Straight: " + player.scoreBoard.ls

+ "\n[ 12 ] Yacht: " + player.scoreBoard.yacht + "\n"

+ "\nTotal: " + player.scoreBoard.total).replace(/null/g, "");

}

};

function onCommand(msg) {

/*if (msg.command == "eval") msg.reply(eval(msg.args.join(" ")));*/

if (msg.command == "y") {

if (msg.args[0] == "help") {

msg.reply("[Yacht!]\n클릭하여 도움말 보기"

+ ("\u200b".repeat(500)) + "\n\n"

+ "!!y join\n- 방 만들기 / 참가\n"

+ "!!y start\n- 게임 시작\n"

+ "!!y roll\n- 주사위 굴리기\n"

+ "!!y keep (주사위눈)...\n- 주사위 보관\n"

+ "!!y disc (주사위눈)...\n- 주사위 꺼내기\n"

+ "!!y rec (족보번호)...\n- 점수 기록\n"

+ "!!y gu\n- 게임 포기");

}

if (msg.args[0] == "join" && !Yacht.started) {

if (!Yacht.made) {

Yacht.init();

msg.reply("[Yacht!]\n" + msg.author.name + "님이 방을 만들었습니다.\n게임에 참가하실 다른 분들은 '!!y join' 으로 참가하세요.");

Yacht.maker = msg.author.name;

Yacht.players.push(new Yacht.Player(msg.author.name));

Yacht.made = true;

} else {

if (Yacht.players.length >= 4) {

msg.reply("[Yacht!]\n최대 4명까지 참가하실 수 있습니다.");

return;

}

msg.reply("[Yacht!]\n" + msg.author.name + "님이 참가하셨습니다.");

Yacht.players.push(new Yacht.Player(msg.author.name));

}

}

if (msg.args[0] == "start" && Yacht.made && msg.author.name == Yacht.maker) {

msg.reply("[Yacht!]\n시작되었습니다.\n" + Yacht.players[0].name + "님, '!!y roll' 로 주사위를 굴려주세요.");

Yacht.started = true;

Yacht.made = false;

}

if (Yacht.started) {

if (msg.args[0] == "gu") {

let guPlayerIndex = Yacht.players.findIndex(_ => _.name == msg.author.name);

if (guPlayerIndex == -1) {

return;

}

if (Yacht.players.length == 1) {

msg.reply("[Yacht!]\n모든 플레이어가 게임을 포기하여 게임이 종료됩니다.");

Yacht.init();

return;

}

Yacht.players.splice(guPlayerIndex, 1);

if (Yacht.players.length == Yacht.turn) {

Yacht.turn = 0;

}

msg.reply("[Yacht!]\n게임을 포기하셨습니다.\n" + Yacht.players[turn] + "님, 주사위를 굴려주세요.");

}

let nowTurnPlayer = Yacht.players[Yacht.turn];

if (msg.author.name != nowTurnPlayer.name) return;

if (msg.args[0] == "roll") {

Yacht.rollDice(nowTurnPlayer);

msg.reply("[Yacht!]\n" + Yacht.printNowDice(nowTurnPlayer));

}

if (msg.args[0] == "keep") {

let keepDice = msg.args.slice(1).map(_ => Math.floor(Number(_)));

if (keepDice <= 0 || keepDice.length > nowTurnPlayer.diceResult.length

|| keepDice.filter(isNaN).length != 0) {

msg.reply("[Yacht!]\n올바른 요청이 아닙니다.");

return;

}

Yacht.keepDice(nowTurnPlayer, keepDice);

msg.reply("[Yacht!]\n" + Yacht.printNowDice(nowTurnPlayer));

}

if (msg.args[0] == "disc") {

let discDice = msg.args.slice(1).map(_ => Math.floor(Number(_)));

if (discDice <= 0 || discDice.length > nowTurnPlayer.diceKeeper.length

|| discDice.filter(isNaN).length != 0) {

msg.reply("[Yacht!]\n올바른 요청이 아닙니다.");

return;

}

Yacht.discardDice(nowTurnPlayer, discDice);

msg.reply("[Yacht!]\n" + Yacht.printNowDice(nowTurnPlayer));

}

if (msg.args[0] == "rec") {

let scoreInd = Math.floor(Number(msg.args[1]));

if (isNaN(scoreInd) || scoreInd < 1 || scoreInd > 12) {

msg.reply("[Yacht!]\n올바른 요청이 아닙니다.");

return;

}

let res = Yacht.recodeDices(nowTurnPlayer, scoreInd);

if (!res) {

msg.reply("[Yacht!]\n이미 기록된 족보입니다.\n다른 족보를 선택하세요.");

return;

}

let nextTurn = Yacht.turn + 1;

if (nextTurn == Yacht.players.length) {

Yacht.realTurn++;

if (Yacht.realTurn == 12) {

Yacht.players.sort((a, b) => b.scoreBoard.total - a.scoreBoard.total);

let yachtResult = "";

for (let i = 0; i < Yacht.players.length; i++) {

yachtResult += "\n" + (i + 1) + "등: " + Yacht.players[i].scoreBoard.total + "점 (플레이어: " + Yacht.players[i].name + ")";

}

msg.reply("[Yacht!]\n모든 턴이 끝났습니다!" + yachtResult);

Yacht.init();

return;

}

nextTurn = 0;

}

msg.reply("[Yacht!]\n" + nowTurnPlayer.name + "님, 기록되었습니다.\n" + Yacht.players[nextTurn].name + "님, 주사위를 굴려주세요.");

Yacht.turn = nextTurn;

}

}

}

}
